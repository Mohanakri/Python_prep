```python
import random

def print_static_poem():
    """Prints the original poem with fixed indentation."""
    print("Twinkle, twinkle, little star,")
    print("\tHow I wonder what you are!")
    print("\t\tUp above the world so high,")
    print("\t\tLike a diamond in the sky.")
    print("Twinkle, twinkle, little star,")
    print("\tHow I wonder what you are!")
    print()

def print_random_indented_poem(poem_lines):
    """Prints each line with random spacing and indentation."""
    print("üé® Randomly Indented Poem:\n")
    for line in poem_lines:
        tabs = "\t" * random.randint(0, 2)
        spaces = " " * random.randint(0, 8)
        print(f"{tabs}{spaces}{line}")
    print()

def format_poem_with_tabs(poem_text):
    """Converts plain text into a tab-indented poem format."""
    print("üßæ Tab-Formatted Poem:\n")
    lines = poem_text.strip().split("\n")
    for i, line in enumerate(lines):
        indent_level = i % 3  # cycles through 0,1,2 for variation
        print("\t" * indent_level + line.strip())
    print()

def main():
    # 1. Print static predefined poem with fixed indentation
    print("üìú Static Indented Poem:\n")
    print_static_poem()

    # 2. Poem lines as list (can be replaced with user input too)
    poem_lines = [
        "The moon shines bright,",
        "Stars twinkle in the night,",
        "Waves whisper to the shore,",
        "As silence sings evermore.",
        "Dreams take flight,",
        "In the calm of midnight."
    ]

    # 3. Print with random indentation
    print_random_indented_poem(poem_lines)

    # 4. Dynamic plain-text poem input (can be replaced with user input)
    plain_text_poem = """
The sun sets low,
Casting golden glow.
Clouds drift slow,
And breezes softly blow.
"""

    # 5. Format plain-text poem using tabs
    format_poem_with_tabs(plain_text_poem)

if __name__ == "__main__":
    main()
```
